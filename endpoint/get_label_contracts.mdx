---
title: "Get Smart Contract Labels"
api: "GET https://walletlabels.xyz/api/sc_names"
description: "This endpoint gets labels for over 200k verified smart contracts."
---

### Body

<ParamField body="query" type="string" required>
  Smart contract address or label to search for.
</ParamField>

<ParamField body="limit" type="integer" optional>
  The maximum number of results to return (default is 20, maximum is 100).
</ParamField>

<ParamField body="API Key" type="integer" optional>
  API Key (optional)
</ParamField>

### Response

<ResponseField name="data" type="array">
  A list of results that match the given address or label entry.

<Expandable title="Label Object">

<ResponseField name="address_name" type="string">
  The name of the smart contract address for the label.
</ResponseField>

{" "}

<ResponseField name="address" type="string">
  The smart contract address
</ResponseField>

{" "}

<ResponseField name="label_type" type="string">
  The type of the label (e.g. contract, token, etc).
</ResponseField>

{" "}

<ResponseField name="label_subtype" type="string">
  The subtype of label (e.g. decentralized exchange, token, marketplace etc).
</ResponseField>

{" "}

<ResponseField name="label" type="string">
  The name of the smart contract. (e.g. Uniswap V2 Factory Contract)
</ResponseField>

  </Expandable>
</ResponseField>
